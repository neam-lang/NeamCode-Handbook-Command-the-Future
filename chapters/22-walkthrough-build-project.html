The Project #

We are going to build BookShelf API — a RESTful API for managing a personal book collection. Users can add books, track reading progress, write notes, and get reading statistics.

Tech stack: Node.js, Express, SQLite, Jest

Why this project: It is small enough to build in one session but complex enough to demonstrate real patterns: CRUD operations, input validation, error handling, testing, and documentation.

Created: src/database.js

  import Database from 'better-sqlite3';
  import { join, dirname } from 'path';
  import { fileURLToPath } from 'url';

  const __dirname = dirname(fileURLToPath(import.meta.url));

  let db;

  export function getDatabase(dbPath) {
    if (db) return db;

    const path = dbPath || join(__dirname, '..', 'bookshelf.db');
    db = new Database(path);

    // Enable WAL mode for better concurrent read performance
    db.pragma('journal_mode = WAL');
    db.pragma('foreign_keys = ON');

    // Create tables
    db.exec(`
      CREATE TABLE IF NOT EXISTS books (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        title TEXT NOT NULL,
        author TEXT NOT NULL,
        isbn TEXT UNIQUE,
        total_pages INTEGER,
        current_page INTEGER DEFAULT 0,
        status TEXT DEFAULT 'want-to-read'
          CHECK (status IN ('want-to-read', 'reading', 'finished')),
        rating INTEGER CHECK (rating IS NULL OR (rating BETWEEN 1 AND 5)),
        started_at DATETIME,
        finished_at DATETIME,
        created_at DATETIME DEFAULT CURRENT_TIMESTAMP
      );

      CREATE TABLE IF NOT EXISTS notes (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        book_id INTEGER NOT NULL REFERENCES books(id) ON DELETE CASCADE,
        content TEXT NOT NULL,
        page_number INTEGER,
        created_at DATETIME DEFAULT CURRENT_TIMESTAMP
      );

      CREATE INDEX IF NOT EXISTS idx_books_status ON books(status);
      CREATE INDEX IF NOT EXISTS idx_notes_book_id ON notes(book_id);
    `);

    return db;
  }

  export function closeDatabase() {
    if (db) {
      db.close();
      db = null;
    }
  }

The dbPath parameter makes testing easier — we can pass ':memory:'
for test databases. WAL mode allows concurrent reads while writing.

Created: src/routes/books.js

  import { Router } from 'express';
  import { getDatabase } from '../database.js';
  import { validateBook, validateBookUpdate } from '../validators/book.validator.js';

  const router = Router();

  // GET /api/books — List all books, optionally filter by status
  router.get('/', (req, res) => {
    const db = getDatabase();
    const { status, search } = req.query;

    let query = 'SELECT * FROM books';
    const params = [];
    const conditions = [];

    if (status) {
      conditions.push('status = ?');
      params.push(status);
    }
    if (search) {
      conditions.push('(title LIKE ? OR author LIKE ?)');
      params.push(`%${search}%`, `%${search}%`);
    }
    if (conditions.length > 0) {
      query += ' WHERE ' + conditions.join(' AND ');
    }
    query += ' ORDER BY created_at DESC';

    const books = db.prepare(query).all(...params);
    res.json({ books, count: books.length });
  });

  // GET /api/books/:id — Get a single book
  router.get('/:id', (req, res) => {
    const db = getDatabase();
    const book = db.prepare('SELECT * FROM books WHERE id = ?').get(req.params.id);

    if (!book) {
      return res.status(404).json({ error: 'Book not found' });
    }
    res.json(book);
  });

  // POST /api/books — Add a new book
  router.post('/', (req, res) => {
    const validation = validateBook(req.body);
    if (!validation.success) {
      return res.status(400).json({ error: validation.error.issues });
    }

    const db = getDatabase();
    const { title, author, isbn, total_pages, status } = validation.data;

    try {
      const result = db.prepare(`
        INSERT INTO books (title, author, isbn, total_pages, status)
        VALUES (?, ?, ?, ?, ?)
      `).run(title, author, isbn || null, total_pages || null, status || 'want-to-read');

      const book = db.prepare('SELECT * FROM books WHERE id = ?').get(result.lastInsertRowid);
      res.status(201).json(book);
    } catch (err) {
      if (err.message.includes('UNIQUE constraint failed: books.isbn')) {
        return res.status(409).json({ error: 'A book with this ISBN already exists' });
      }
      throw err;
    }
  });

  // PUT /api/books/:id — Update a book
  router.put('/:id', (req, res) => {
    const db = getDatabase();
    const existing = db.prepare('SELECT * FROM books WHERE id = ?').get(req.params.id);
    if (!existing) {
      return res.status(404).json({ error: 'Book not found' });
    }

    const validation = validateBookUpdate(req.body);
    if (!validation.success) {
      return res.status(400).json({ error: validation.error.issues });
    }

    const data = validation.data;
    const fields = Object.keys(data).map(k => `${k} = ?`).join(', ');
    const values = Object.values(data);

    db.prepare(`UPDATE books SET ${fields} WHERE id = ?`).run(...values, req.params.id);
    const book = db.prepare('SELECT * FROM books WHERE id = ?').get(req.params.id);
    res.json(book);
  });

  // DELETE /api/books/:id — Delete a book
  router.delete('/:id', (req, res) => {
    const db = getDatabase();
    const result = db.prepare('DELETE FROM books WHERE id = ?').run(req.params.id);

    if (result.changes === 0) {
      return res.status(404).json({ error: 'Book not found' });
    }
    res.status(204).send();
  });

  // PATCH /api/books/:id/progress — Update reading progress
  router.patch('/:id/progress', (req, res) => {
    const db = getDatabase();
    const book = db.prepare('SELECT * FROM books WHERE id = ?').get(req.params.id);

    if (!book) {
      return res.status(404).json({ error: 'Book not found' });
    }

    const { current_page } = req.body;
    if (typeof current_page !== 'number' || current_page < 0) {
      return res.status(400).json({ error: 'current_page must be a non-negative number' });
    }
    if (book.total_pages && current_page > book.total_pages) {
      return res.status(400).json({ error: 'current_page cannot exceed total_pages' });
    }

    // Auto-update status based on progress
    let status = book.status;
    let started_at = book.started_at;
    let finished_at = book.finished_at;

    if (current_page > 0 && status === 'want-to-read') {
      status = 'reading';
      started_at = new Date().toISOString();
    }
    if (book.total_pages && current_page >= book.total_pages) {
      status = 'finished';
      finished_at = new Date().toISOString();
    }

    db.prepare(`
      UPDATE books
      SET current_page = ?, status = ?, started_at = ?, finished_at = ?
      WHERE id = ?
    `).run(current_page, status, started_at, finished_at, req.params.id);

    const updated = db.prepare('SELECT * FROM books WHERE id = ?').get(req.params.id);
    res.json(updated);
  });

  export default router;

Note the progress endpoint auto-updates status:
  - Setting current_page > 0 auto-sets status to "reading"
  - Setting current_page = total_pages auto-sets to "finished"
  - Timestamps are set automatically too

Created: src/routes/stats.js

  import { Router } from 'express';
  import { getDatabase } from '../database.js';

  const router = Router();

  // GET /api/stats
  router.get('/', (req, res) => {
    const db = getDatabase();

    const totalBooks = db.prepare('SELECT COUNT(*) as count FROM books').get();
    const byStatus = db.prepare(
      'SELECT status, COUNT(*) as count FROM books GROUP BY status'
    ).all();

    const totalPages = db.prepare(
      'SELECT COALESCE(SUM(current_page), 0) as total FROM books'
    ).get();

    const thisMonth = new Date();
    thisMonth.setDate(1);
    thisMonth.setHours(0, 0, 0, 0);

    const finishedThisMonth = db.prepare(
      'SELECT COUNT(*) as count FROM books WHERE status = ? AND finished_at >= ?'
    ).get('finished', thisMonth.toISOString());

    const averageRating = db.prepare(
      'SELECT ROUND(AVG(rating), 1) as avg FROM books WHERE rating IS NOT NULL'
    ).get();

    const topAuthors = db.prepare(
      'SELECT author, COUNT(*) as book_count FROM books GROUP BY author ORDER BY book_count DESC LIMIT 5'
    ).all();

    res.json({
      total_books: totalBooks.count,
      books_by_status: Object.fromEntries(byStatus.map(r => [r.status, r.count])),
      total_pages_read: totalPages.total,
      books_finished_this_month: finishedThisMonth.count,
      average_rating: averageRating.avg,
      top_authors: topAuthors
    });
  });

  export default router;

Created: src/validators/book.validator.js

  import { z } from 'zod';

  const isbnRegex = /^(?:\d{10}|\d{13}|\d{1,5}-\d{1,7}-\d{1,7}-[\dX])$/;

  const bookSchema = z.object({
    title: z.string().min(1, 'Title is required').max(500),
    author: z.string().min(1, 'Author is required').max(200),
    isbn: z.string().regex(isbnRegex, 'Invalid ISBN format').optional(),
    total_pages: z.number().int().positive().max(50000).optional(),
    status: z.enum(['want-to-read', 'reading', 'finished']).optional(),
    rating: z.number().int().min(1).max(5).optional(),
  });

  const bookUpdateSchema = z.object({
    title: z.string().min(1).max(500).optional(),
    author: z.string().min(1).max(200).optional(),
    isbn: z.string().regex(isbnRegex, 'Invalid ISBN format').optional(),
    total_pages: z.number().int().positive().max(50000).optional(),
    status: z.enum(['want-to-read', 'reading', 'finished']).optional(),
    rating: z.number().int().min(1).max(5).nullable().optional(),
    current_page: z.number().int().min(0).optional(),
  }).refine(data => Object.keys(data).length > 0, {
    message: 'At least one field must be provided for update',
  });

  export function validateBook(data) {
    return bookSchema.safeParse(data);
  }

  export function validateBookUpdate(data) {
    return bookUpdateSchema.safeParse(data);
  }

Notes:
  - ISBN accepts both ISBN-10 and ISBN-13 formats
  - Rating is nullable on update (to clear a rating)
  - Update requires at least one field (can't send empty body)
  - total_pages max is 50,000 (covers even Proust)

  export function errorHandler(err, req, res, next) {
    console.error(`[ERROR] ${req.method} ${req.path}:`, err.message);

    if (err.type === 'entity.parse.failed') {
      return res.status(400).json({ error: 'Invalid JSON in request body' });
    }

    res.status(500).json({
      error: 'Internal server error',
      ...(process.env.NODE_ENV === 'development' && { details: err.message }),
    });
  }

  export function notFoundHandler(req, res) {
    res.status(404).json({
      error: `Route ${req.method} ${req.path} not found`,
    });
  }

  import express from 'express';
  import cors from 'cors';
  import morgan from 'morgan';
  import bookRoutes from './routes/books.js';
  import noteRoutes from './routes/notes.js';
  import statsRoutes from './routes/stats.js';
  import { errorHandler, notFoundHandler } from './middleware/errorHandler.js';

  export function createApp() {
    const app = express();

    // Middleware
    app.use(cors());
    app.use(express.json());
    if (process.env.NODE_ENV !== 'test') {
      app.use(morgan('dev'));
    }

    // Health check
    app.get('/health', (req, res) => {
      res.json({ status: 'ok', timestamp: new Date().toISOString() });
    });

    // Routes
    app.use('/api/books', bookRoutes);
    app.use('/api', noteRoutes);
    app.use('/api/stats', statsRoutes);

    // Error handling
    app.use(notFoundHandler);
    app.use(errorHandler);

    return app;
  }

# BookShelf API

A RESTful API for managing your personal book collection.
Track reading progress, add notes, and view statistics.

## Quick Start

  git clone <repo-url>

  # List all books (with pagination)
  curl http://localhost:3000/api/books?page=1&limit=20

  # Filter by status
  curl http://localhost:3000/api/books?status=reading

  # Search by title or author
  curl http://localhost:3000/api/books?search=Dune

  # Get a single book
  curl http://localhost:3000/api/books/1

  # Add a book
  curl -X POST http://localhost:3000/api/books \
    -H "Content-Type: application/json" \
    -d '{"title":"Dune","author":"Frank Herbert","total_pages":412}'

  # Update a book
  curl -X PUT http://localhost:3000/api/books/1 \
    -H "Content-Type: application/json" \
    -d '{"rating":5}'

  # Update reading progress
  curl -X PATCH http://localhost:3000/api/books/1/progress \
    -H "Content-Type: application/json" \
    -d '{"current_page":150}'

  # Delete a book
  curl -X DELETE http://localhost:3000/api/books/1

    -d '{"content":"Great worldbuilding on page 42","page_number":42}'

  # Delete a note
  curl -X DELETE http://localhost:3000/api/notes/1

### Statistics

  # Get reading stats
